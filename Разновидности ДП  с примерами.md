```cpp
int gridPaths(int n, int m) {
    vector<vector<int>> dp(n, vector<int>(m, 1));
    for (int i = 1; i < n; i++) {
        for (int j = 1; j < m; j++) {
            dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
        }
    }
    return dp[n - 1][m - 1];
}
```

```cpp
int LCS(string a, string b) {
    int n = a.size(), m = b.size();
    vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            if (a[i - 1] == b[j - 1])
                dp[i][j] = dp[i - 1][j - 1] + 1;
            else
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
        }
    }
    return dp[n][m];
}
```

### –ü—Ä–∏–º–µ—Ä: –ö–æ–º–º–∏–≤–æ—è–∂—ë—Ä (TSP, Traveling Salesman Problem)
```cpp
const int INF = 1e9;
vector<vector<int>> dist;
vector<vector<int>> dp;
int tsp(int mask, int pos, int n) {
    if (mask == (1 << n) - 1) return dist[pos][0];
    if (dp[mask][pos] != -1) return dp[mask][pos];
    int res = INF;
    for (int next = 0; next < n; next++) {
        if (!(mask & (1 << next))) {
            res = min(res, dist[pos][next] + tsp(mask | (1 << next), next, n));
        }
    }
    return dp[mask][pos] = res;
}
```

```cpp
    int maxSum = 0;
    for (int child : tree[v]) {
        if (child == parent) continue;

}
```

```cpp
int groupPartition(vector<int>& arr, int k) {
    int n = arr.size();
    vector<vector<int>> dp(n + 1, vector<int>(k + 1, 1e9));
    dp[0][0] = 0;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= k; j++) {
            int sum = 0;
            for (int p = i; p >= 1; p--) {
                sum += arr[p - 1];
                dp[i][j] = min(dp[i][j], max(dp[p - 1][j - 1], sum));
            }
        }
    }
    return dp[n][k];
}
```
